🧠 八、模型开发与训练（PyTorch）

模型定义

import torch.nn as nn

class MyModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.linear = nn.Linear(768, 2)
    
    def forward(self, x):
        return self.linear(x)
